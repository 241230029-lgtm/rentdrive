<?php

namespace App\Http\Controllers;

use App\Models\Kendaraan;
use App\Models\Sewa;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Auth;
use Illuminate\Support\Facades\Storage;

    /**
     * 1. ALUR PELANGGAN: Memproses pengajuan booking online dari pelanggan
     * Validasi anti-bentrok jadwal berjalan secara rahasia di sini.
     */
    public function simpanBookingOnline(Request $request)
    {
        // Validasi Input Tanggal Sesuai Aturan Bisnis
        $request->validate([
            'kendaraan_id'  => 'required|exists:kendaraans,id',
            'tanggal_mulai' => 'required|date|after_or_equal:today',
            'tanggal_selesai'=> 'required|date|after:tanggal_mulai',
        ]);

        $inputMulai = $request->tanggal_mulai;
        $inputSelesai = $request->tanggal_selesai;

        // Ambil data spesifikasi kendaraan
        $kendaraan = Kendaraan::findOrFail($request->kendaraan_id);

        // Proteksi awal: Cek jika status unit dinonaktifkan secara manual oleh admin
        if (in_array($kendaraan->status, ['perbaikan', 'tidak_aktif'])) {
            return back()->with('error', 'Maaf, kendaraan saat ini sedang tidak dapat digunakan untuk operasional.');
        }

        // LOGIKA RAHASIA BACKEND: Menghitung sisa unit tersedia pada rentang tanggal terpilih
        $jumlahUnitTerpakai = Sewa::where('kendaraan_id', $kendaraan->id)
            ->whereNotIn('status', ['dibatalkan', 'ditolak_pembayaran', 'selesai'])
            ->where(function ($query) use ($inputMulai, $inputSelesai) {
                $query->where('tanggal_mulai', '<=', $inputSelesai)
                      ->where('tanggal_selesai', '>=', $inputMulai);
            })
            ->count();

        // Tolak jika kuota pemesanan online + offline sudah memenuhi kapasitas unit fisik perusahaan
        if ($jumlahUnitTerpakai >= $kendaraan->jumlah_unit) {
            return back()->with('error', 'Maaf, kendaraan tidak tersedia pada rentang tanggal yang Anda pilih. Silakan pilih jadwal lain.');
        }

        // Hitung Masa Durasi Sewa & Akumulasi Biaya
        $mulai = Carbon::parse($inputMulai);
        $selesai = Carbon::parse($inputSelesai);
        $durasiHari = $mulai->diffInDays($selesai);

        $totalHarga = $durasiHari * $kendaraan->harga_per_hari;

        // Otomatisasi nomor booking sementara (Audit Trail aman)
        $nomorBooking = 'BK-' . Carbon::now()->format('Ymd') . '-' . strtoupper(Str::random(5));

        // Buat data dengan status awal 'menunggu_pembayaran'
        Sewa::create([
            'nomor_booking'   => $nomorBooking,
            'user_id'         => Auth::id(), // ðŸ‘ˆ Sudah diperbaiki menggunakan Facade (Bebas Eror Intelephense)
            'kendaraan_id'    => $kendaraan->id,
            'jenis_booking'   => 'online',
            'tanggal_mulai'   => $inputMulai,
            'tanggal_selesai' => $inputSelesai,
            'total_harga'     => $totalHarga,
            'status'          => 'menunggu_pembayaran',
        ]);

        return redirect()->route('pelanggan.dashboard')->with('success', 'Booking berhasil dibuat! Nomor Pemesanan Anda: ' . $nomorBooking . '. Silakan lakukan unggah bukti pembayaran.');
    }

    /**
     * 2. ALUR PELANGGAN: Mengunggah bukti pembayaran (Dummy Payment)
     * Mengubah status dari 'menunggu_pembayaran' atau 'ditolak_pembayaran' menjadi 'menunggu_verifikasi_pembayaran'
     */
    public function unggahBuktiPembayaran(Request $request, int $id)
    {
        // Validasi File Bukti Transfer (Jenis file, ukuran maksimal 2MB)
        $request->validate([
            'bukti_pembayaran' => 'required|image|mimes:jpeg,png,jpg|max:2048',
        ]);

        // Pastikan transaksi ini benar-benar milik pelanggan yang sedang login
        $sewa = Sewa::where('id', $id)->where('user_id', Auth::id())->firstOrFail();

        // Ambil file dari request
        if ($request->hasFile('bukti_pembayaran')) {

            // Hapus file bukti lama jika pelanggan melakukan unggah ulang akibat penolakan sebelumnya
            if ($sewa->bukti_pembayaran) {
                Storage::disk('public')->delete($sewa->bukti_pembayaran);
            }

            // Simpan file ke dalam folder storage/app/public/bukti_pembayaran
            $namaFile = $request->file('bukti_pembayaran')->store('bukti_pembayaran', 'public');

            // Perbarui status transaksi di database menuju antrean verifikasi Admin
            $sewa->update([
                'bukti_pembayaran' => $namaFile,
                'status' => 'menunggu_verifikasi_pembayaran'
            ]);

            return redirect()->route('pelanggan.dashboard')->with('success', 'Bukti pembayaran berhasil diunggah! Menunggu proses verifikasi oleh Administrator.');
        }

        return back()->with('error', 'Gagal memproses file bukti pembayaran. Silakan coba kembali.');
    }

     * Langsung disetujui operasional dan dicatat ke log aktivitas untuk audit Owner.

    public function simpanBookingWalkIn(Request $request)
    {
        // Validasi input data penyewa offline dan jadwal kontrak sewa
        $request->validate([
            'pelanggan_id'  => 'required|exists:users,id', // Pelanggan dipilih dari daftar user terdaftar
            'kendaraan_id'  => 'required|exists:kendaraans,id',
            'tanggal_mulai' => 'required|date|after_or_equal:today',
            'tanggal_selesai'=> 'required|date|after:tanggal_mulai',
        ]);

        $inputMulai = $request->tanggal_mulai;
        $inputSelesai = $request->tanggal_selesai;

        // Ambil data spesifikasi kendaraan
        $kendaraan = Kendaraan::findOrFail($request->kendaraan_id);

        // Proteksi awal: Cek jika unit dalam perbaikan atau non-aktif
        if (in_array($kendaraan->status, ['perbaikan', 'tidak_aktif'])) {
            return back()->with('error', 'Gagal. Kendaraan saat ini sedang dalam masa perbaikan atau tidak aktif.');
        }

        // LOGIKA RAHASIA BACKEND: Cek bentrok jadwal kuota multi-unit fisik perusahaan
        $jumlahUnitTerpakai = Sewa::where('kendaraan_id', $kendaraan->id)
            ->whereNotIn('status', ['dibatalkan', 'ditolak_pembayaran', 'selesai'])
            ->where(function ($query) use ($inputMulai, $inputSelesai) {
                $query->where('tanggal_mulai', '<=', $inputSelesai)
                      ->where('tanggal_selesai', '>=', $inputMulai);
            })
            ->count();

        if ($jumlahUnitTerpakai >= $kendaraan->jumlah_unit) {
            return back()->with('error', 'Gagal. Kendaraan tidak tersedia pada rentang tanggal tersebut karena kuota unit fisik sudah penuh.');
        }

        // Kalkulasi Durasi Hari & Total Nominal Pembayaran Tunai
        $mulai = Carbon::parse($inputMulai);
        $selesai = Carbon::parse($inputSelesai);
        $durasiHari = $mulai->diffInDays($selesai);
        $totalHarga = $durasiHari * $kendaraan->harga_per_hari;

        // Generate Nomor Booking Manual Offline
        $nomorBooking = 'WI-' . Carbon::now()->format('Ymd') . '-' . strtoupper(Str::random(5));

        // Buat data transaksi dengan status langsung 'disetujui_operasional'
        $sewa = Sewa::create([
            'nomor_booking'   => $nomorBooking,
            'user_id'         => $request->pelanggan_id, // Terikat ke user pelanggan pilihan admin
            'kendaraan_id'    => $kendaraan->id,
            'jenis_booking'   => 'walk_in', // Penanda jenis transaksi offline
            'tanggal_mulai'   => $inputMulai,
            'tanggal_selesai' => $inputSelesai,
            'total_harga'     => $totalHarga,
            'bukti_pembayaran'=> 'WALK_IN_CASH', // Penanda dibayar tunai di kasir
            'status'          => 'disetujui_operasional',
        ]);

        // AMAN UNTUK OWNER: Catat log aktivitas tindakan admin
        $pelanggan = \App\Models\User::find($request->pelanggan_id);
        \App\Models\LogAktivitas::create([
            'user_id' => Auth::id(), // ID Admin pengeksekusi transaksi offline
            'jenis_aktivitas' => 'Input Booking Walk-In',
            'deskripsi' => 'Admin membuat booking Walk-In langsung untuk Nomor ' . $nomorBooking . ' (Pelanggan: ' . $pelanggan->name . ', Kendaraan: ' . $kendaraan->nama_kendaraan . ')',
            'alamat_ip' => $request->ip(),
        ]);

        return redirect()->route('admin.booking.index')->with('success', 'Transaksi Walk-In berhasil dicatat dengan Nomor: ' . $nomorBooking . '. Status operasional aktif.');
    }

    /**
     * 5. ALUR ADMINISTRATOR: Penyelesaian Pengembalian Kendaraan & Hitung Otomatis Denda
     * Mengubah status transaksi menjadi 'selesai' dan mencatat log audit untuk Owner.
     */
    public function selesaikanPengembalianAdmin(Request $request, int $id)
    {
        // 1. Validasi input pemeriksaan fisik dari Admin di lapangan
        $request->validate([
            'tanggal_kembali_aktual'    => 'required|date|after_or_equal:tanggal_mulai',
            'kondisi_kendaraan_kembali' => 'required|string',
            'foto_kondisi_kembali'      => 'nullable|image|mimes:jpeg,png,jpg|max:2048',
            'kilometer_kembali'         => 'required|integer',
            'denda_kerusakan'           => 'required|integer|min:0', // Diinput manual oleh admin berdasarkan lecet fisik
        ]);

        // Ambil data transaksi sewa beserta relasinya
        $sewa = Sewa::with('kendaraan', 'user')->findOrFail($id);

        // 2. LOGIKA OTOMATISASI: Hitung Denda Keterlambatan
        $tanggalSelesaiKontrak = Carbon::parse($sewa->tanggal_selesai);
        $tanggalKembaliAktual  = Carbon::parse($request->tanggal_kembali_aktual);

        $dendaKeterlambatan = 0;

        // Jika tanggal kembali aktual melewati tanggal selesai kontrak sewa
        if ($tanggalKembaliAktual->gt($tanggalSelesaiKontrak)) {
            $hariKeterlambatan = $tanggalKembaliAktual->diffInDays($tanggalSelesaiKontrak);
            // Denda per hari = harga sewa per hari kendaraan tersebut
            $dendaKeterlambatan = $hariKeterlambatan * $sewa->kendaraan->harga_per_hari;
        }

        // 3. Akumulasi Total Denda
        $dendaKerusakan = $request->denda_kerusakan;
        $totalDenda = $dendaKeterlambatan + $dendaKerusakan;

        // 4. Proses Upload Foto Kondisi Fisik Mobil Saat Dikembalikan (Jika Ada)
        $namaFotoKembali = $sewa->foto_kondisi_kembali;
        if ($request->hasFile('foto_kondisi_kembali')) {
            $namaFotoKembali = $request->file('foto_kondisi_kembali')->store('kondisi_kembali', 'public');
        }

        // 5. Perbarui Data Transaksi dan Ubah Status Menjadi Selesai
        $sewa->update([
            'tanggal_kembali_aktual'    => $request->tanggal_kembali_aktual,
            'kondisi_kendaraan_kembali' => $request->kondisi_kendaraan_kembali,
            'foto_kondisi_kembali'      => $namaFotoKembali,
            'kilometer_kembali'         => $request->kilometer_kembali,
            'denda_keterlambatan'       => $dendaKeterlambatan,
            'denda_kerusakan'           => $dendaKerusakan,
            'total_denda'               => $totalDenda,
            'status'                    => 'selesai', // Siklus sewa dinyatakan ditutup
        ]);

        // 6. AMAN UNTUK OWNER: Catat log aktivitas tindakan penyelesaian oleh admin
        \App\Models\LogAktivitas::create([
            'user_id' => Auth::id(), // ID Admin penanggung jawab lapangan
            'jenis_aktivitas' => 'Penyelesaian Pengembalian',
            'deskripsi' => 'Admin menyelesaikan transaksi ' . $sewa->nomor_booking . '. Total Denda: Rp ' . number_format($totalDenda, 0, ',', '.') . ' (Keterlambatan: Rp ' . number_format($dendaKeterlambatan, 0, ',', '.') . ', Kerusakan: Rp ' . number_format($dendaKerusakan, 0, ',', '.') . ')',
            'alamat_ip' => $request->ip(),
        ]);

        return redirect()->route('admin.pengembalian.index')->with('success', 'Proses pengembalian berhasil diproses. Transaksi ' . $sewa->nomor_booking . ' dinyatakan Selesai.');
    }
}

        'denda_kerusakan',
        'total_denda',
        'alasan_penolakan',
        'status',
    ];

    /**
     * Relasi: Transaksi sewa ini DIMILIKI OLEH seorang User (Pelanggan).

import { Head, Link } from '@inertiajs/react';

export default function RiwayatSewa({ riwayatSewa = [] }) {
    const daftarRiwayat = Array.isArray(riwayatSewa)
        ? riwayatSewa
        : [];

    const formatTanggal = (tanggal) => {
        if (!tanggal) {
            return '-';
        }

        return new Intl.DateTimeFormat('id-ID', {
            day: '2-digit',
            month: 'long',
            year: 'numeric',
        }).format(new Date(tanggal));
    };

    const kelasStatus = (status) => {
        const nilai = status?.toLowerCase();

        if (
            nilai === 'diterima' ||
            nilai === 'disetujui' ||
            nilai === 'selesai'
        ) {
            return 'border-emerald-500/30 bg-emerald-500/10 text-emerald-300';
        }

        if (nilai === 'ditolak' || nilai === 'dibatalkan') {
            return 'border-rose-500/30 bg-rose-500/10 text-rose-300';
        }

        return 'border-amber-500/30 bg-amber-500/10 text-amber-300';
    };

    return (
        <>
            <Head title="Riwayat Sewa" />

            <main className="mx-auto max-w-7xl px-5 py-10 sm:px-6 lg:px-8">
                <section className="rounded-3xl border border-slate-800 bg-[#10192B] px-6 py-10 text-center shadow-2xl">
                    <p className="text-xs font-bold uppercase tracking-[0.25em] text-[#06B6D4]">
                        Transaksi Pelanggan
                    </p>

                    <h1 className="mt-4 text-3xl font-extrabold sm:text-4xl">
                        Riwayat Penyewaan
                    </h1>

                    <p className="mx-auto mt-4 max-w-2xl text-sm leading-7 text-[#94A3B8]">
                        Pantau permintaan pemesanan, proses konfirmasi,
                        pembayaran, dan penyelesaian penyewaan kendaraan Anda.
                    </p>
                </section>

                <section className="mt-10">
                    {daftarRiwayat.length === 0 ? (
                        <div className="rounded-2xl border border-slate-800 bg-[#1E293B] px-6 py-16 text-center shadow-xl">
                            <div className="mx-auto flex h-14 w-14 items-center justify-center rounded-xl bg-[#06B6D4]/10">
                                <svg
                                    viewBox="0 0 24 24"
                                    fill="none"
                                    stroke="currentColor"
                                    strokeWidth="1.7"
                                    className="h-7 w-7 text-[#06B6D4]"
                                >
                                    <path
                                        strokeLinecap="round"
                                        strokeLinejoin="round"
                                        d="M6 3h12v18H6zM9 8h6M9 12h6M9 16h4"
                                    />
                                </svg>
                            </div>

                            <h2 className="mt-5 text-xl font-bold">
                                Belum ada riwayat penyewaan
                            </h2>

                            <p className="mx-auto mt-3 max-w-md text-sm leading-7 text-[#94A3B8]">
                                Riwayat transaksi akan muncul setelah Anda
                                mengajukan pemesanan melalui halaman katalog.
                            </p>

                            <Link
                                href={route('pelanggan.katalog')}
                                viewTransition
                                className="mt-6 inline-flex rounded-xl bg-[#06B6D4] px-6 py-3 text-sm font-bold text-[#0B1120] hover:bg-[#0891B2]"
                            >
                                Buka Katalog
                            </Link>
                        </div>
                    ) : (
                        <div className="space-y-5">
                            {daftarRiwayat.map((sewa) => (
                                <article

                                                {formatTanggal(

import './bootstrap';
import '../css/app.css';

import PelangganLayout from '@/Layouts/PelangganLayout';
import { createInertiaApp } from '@inertiajs/react';
import { resolvePageComponent } from 'laravel-vite-plugin/inertia-helpers';
import { createRoot } from 'react-dom/client';

const appName = import.meta.env.VITE_APP_NAME || 'RentDrive';

createInertiaApp({
    title: (title) => `${title} - ${appName}`,

    resolve: async (name) => {
        const page = await resolvePageComponent(
            `./Pages/${name}.jsx`,
            import.meta.glob('./Pages/**/*.jsx'),
        );

        /*
         * Semua halaman pelanggan memakai persistent layout yang sama.
         * Header dan footer tidak dimuat ulang ketika navigasi.
         */
        if (
            name === 'Dashboard' ||
            name.startsWith('Pelanggan/')
        ) {
            page.default.layout =
                page.default.layout ||
                ((pageContent) => (
                    <PelangganLayout>
                        {pageContent}
                    </PelangganLayout>
                ));
        }

        return page;
    },

    setup({ el, App, props }) {
        const root = createRoot(el);

        root.render(<App {...props} />);
    },

    /*
     * Progress baru muncul setelah 250 ms.
     * Navigasi cepat hampir tidak memperlihatkan loading bar.
     */
    progress: {
        delay: 250,
        color: '#06B6D4',
        includeCSS: true,
        showSpinner: false,
    },
});
